import uvicorn
from fastapi import FastAPI, Body, status
from fastapi.responses import JSONResponse
from fastapi.middleware.cors import CORSMiddleware
# pip install PyJWT 설치
import jwt

app = FastAPI()
# CORS 설정
cors_config = {
    "allow_origins":"http://localhost:3000",
    "allow_credentials":True,
    "allow_methods":["*"],
    "allow_headers":["*"]
}
app.add_middleware(
    CORSMiddleware, 
    # allow_origins = "http://localhost:3000"
    **cors_config
    # **: 언패킹 -> 딕셔너리 키값을 옵션명 value값을 옵션값
    # allow_orgins = "http://localhost:3000"
)

# 원래는 파일 따로 만들어야 하나 여기에 임시 테스트 진행(보통 models 폴더나 DB 설정용 별도 모듈에 위치시켜 관리)
from sqlalchemy import create_engine, text

# 이 부분은 데이터베이스 접속에 필요한 정보(사용자 ID, 비밀번호, 호스트, 데이터베이스명)를 문자열 형태로 조합
# 데이터베이스에 연결하기 위해 미리 지정한 계정명, 비밀번호, 호스트 주소, DB 이름을 하드코딩
user_id = "root"
db_password = "test"
host = "localhost:3307"
db_name = "ex260310"
db_info = f"mysql+pymysql://{user_id}:{db_password}@{host}/{db_name}"
print(db_info)
# db_info 변수는 연결에 필요한 데이터베이스 접속 문자열(URI)를 담고 있음.


# 위에서 만든 접속 정보를 바탕으로 SQLAlchemy의 데이터베이스 엔진 객체를 생성하는 코드
# 이 엔진은 Python 백엔드 애플리케이션이 해당 데이터베이스 서버와 통신하고 쿼리를 실행할 수 있도록 하는 연결 통로 역할
engine = create_engine(
    db_info, connect_args={})  # create_engine(db_info)는 이 URI 정보를 바탕으로 DB 서버와 통신할 수 있는 연결 객체(엔진)를 만듦
print(engine)
# print(db_info)와 print(engine)은 각각 접속 문자열과 생성된 엔진 객체 정보를 출력

# 위에 코드까지는 데이터베이스 서버에 먼저 로그인해서 접속 권한을 얻고, 
# 그 이후에 테이블이나 데이터에 접근하는 작업을 하기 위한 준비 과정임
'''from sqlalchemy import create_engine, text 부분은

 MySQL 데이터베이스에 접속하기 위한 사용자 계정과 비밀번호, 호스트 주소, 데이터베이스 이름 정보를 
 연결 문자열(URI)로 만들어 
 SQLAlchemy의 create_engine 함수에 전달하여 
 데이터베이스와 통신하기 위한 "연결 통로" 역할의 엔진 객체를 생성하는 단계
'''

@app.post("/login")
def login(
        token = Body(..., embed=True)
    ):
    print(token)
    
    decoded = jwt.decode(token, options={
        "verify_signature":False}) # 토큰 서명(signature)의 검증을 건너뛴다는 의미
    print(decoded)
    
    query = text("""
    SELECT count(*) as n FROM users 
    WHERE email = :email & name = :name
    """)
    
    # 데이터베이스에 이메일과 이름이 있다는 가정하에 로그인 정보와 데이터 베이스 정보가 같은지 비교하는 코드    
    email = decoded["email"]
    name = decoded["name"]
    with engine.connect() as conn:
            row = conn.execute(
                query, {
                    "email":email, 
                    "name":name}
            ).fetchone()
    print(row)
    
    # 받아온 이메일과 이름의 계정 정보가 없다면 로그인 실패
    if row[0]==0:
        return JSONResponse(
            status_code=status.HTTP_401_UNAUTHORIZED,   # 401 Unauthorized (권한 없음)
            content={"message":"로그인 실패"}
        )        
 
 
# 401 Unauthorized (권한 없음)
# 인증이 필요하거나, 인증 정보가 유효하지 않을 때 사용합니다.
# 즉, 로그인 정보가 없거나 틀려서 인증 실패인 경우 적절합니다.

# 403 Forbidden (접근 금지)
# 인증은 되었으나, 해당 리소스에 접근 권한이 없을 때 사용합니다.
# 예를 들어 로그인은 했지만 역할 권한이 부족할 때 씁니다.
    
    return JSONResponse(
        
        status_code= status.HTTP_200_OK,
        content={"message":"로그인 성공"}
    )
    
if __name__ == "__main__":
    
    uvicorn.run(
        "main:app", host = "0.0.0.0", port = 5000, reload = True
    )
